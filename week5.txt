#include <iostream>
#include <string>
#include <fstream>
using namespace std;

class Item {
    public:
        string name;
        int quantity;
        void saveToFile() {
            ofstream outputFile("items.txt");
            if (outputFile.is_open()) {
                outputFile << name << "," << quantity << endl;
                outputFile.close();
                cout << "File successfully saved." << endl;
            } 
            else {
                cout << "Error: Unable to open file for writing." << endl;
            }
        }

        void loadFromFile() {
            ifstream inputFile("items.txt");
            string line;
            if (inputFile.is_open()) {
                while (getline(inputFile, line)) {
                    cout << "File content: " << line << endl;
                }
                inputFile.close();
                cout << "File successfully loaded." << endl;
            } 
            else {
                cout << "Error: Unable to open file for reading." << endl;
            }
        }
};

int main() {
    Item computer;
    computer.name = "computer";
    computer.quantity = 3;
    computer.saveToFile();
    computer.loadFromFile();
    return 0;
}
